UniversalPicker.js

A lightweight, professional date range picker for the web. Calendar, Accounting Period, and DoubleDate modes — ready to use via CDN.

Date Range Single Date Accounting Periods Double Date Time Picker Predefined Ranges Min/Max Date Locale Support Responsive
Getting Started
Installation
Just one file. No build tools, bundler, or other dependencies needed.
<script src="https://cdn.jsdelivr.net/npm/@farisulhaq/universal-picker@3.0.1/dist/universal-picker.min.js"></script>
npm install @farisulhaq/universal-picker

Live Playground

Date Range Picker
Sidebar ranges, week numbers, dropdowns
Default
No date selected
var picker = new UniversalPicker('#daterange', { mode: 'default', format: 'DD MMM YYYY', showCalendars: 2, showDropdowns: true, showWeekNumbers: true, firstDay: 1, ranges: { 'Hari Ini': [today, today], '7 Hari Terakhir': [daysAgo(6), today], 'Bulan Ini': [startOfMonth(), endOfMonth()], 'Bulan Lalu': [startOfLastMonth(), endOfLastMonth()] }, locale: { applyLabel: 'Terapkan', cancelLabel: 'Batal', customRangeLabel: 'Pilih Sendiri' }, onApply: function (data) { console.log(data.startDate, data.endDate); } });
📆
Single Date Picker
Auto-apply, min/max, disabled weekends
Single
No date selected
var picker = new UniversalPicker('#single', { singleDatePicker: true, autoApply: true, format: 'DD MMMM YYYY', showCalendars: 1, showDropdowns: true, minDate: new Date(2025, 0, 1), maxDate: new Date(2027, 11, 31), // Disable weekends isInvalidDate: function (date) { var day = date.getDay(); return day === 0 || day === 6; }, onApply: function (data) { console.log(data.startDate); } });
📈
Accounting Period
Custom periods + sidebar ranges
Custom
No period selected
var picker = new UniversalPicker('#accounting', { mode: 'custom', format: 'DD MMM YYYY', showCalendars: 2, accountingConfig: [ { label: 'P01 - Jan', start: new Date(2025,11,29), end: new Date(2026,0,25) }, { label: 'P02 - Feb', start: new Date(2026,0,26), end: new Date(2026,1,22) }, // ... more periods ], ranges: { 'Hari Ini': [today, today], 'Bulan Ini': [startOfMonth(), endOfMonth()] }, locale: { customRangeLabel: 'Pilih Periode' // switches back to accounting view } });
🕐
Double Date
Quick pick 1/1, 2/2, ... 12/12
DoubleDate
No date selected
var picker = new UniversalPicker('#doubledate', { mode: 'doubledate', format: 'DD/MM/YYYY', autoApply: true, onApply: function (data) { console.log('Double date:', data.startDate); } });
⏱
Time Picker
Date range with time selection
Time
No date selected
var picker = new UniversalPicker('#datetime', { mode: 'default', format: 'DD/MM/YYYY HH:mm', showCalendars: 2, showDropdowns: true, timePicker: true, timePicker24Hour: true, timePickerIncrement: 15, firstDay: 1, onApply: function (data) { console.log(data.startDate, data.endDate); } });
